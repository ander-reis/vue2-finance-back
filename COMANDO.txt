comando para iniciar docker:
docker-compose up -d

instalar prisma cli:
npm i -g prisma@1.28

iniciar prisma endpoint:
prisma init --endpoint http://localhost:4466

atualiza schema prisma server:
prisma deploy

gera prisma client:
prisma generate

docker-compose up -d --build --no-deps api
